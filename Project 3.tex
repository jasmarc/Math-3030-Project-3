\subsection{Equations} The following equations are used to calculate the mean, variance, and standard deviation of a set of data: 
\newline \textbf{mean} 
\begin{equation}
	\bar{x} = \frac{x_{1} + x_{2} + \cdots + x_{n}}{n} 
\end{equation}
\textbf{variance} 
\begin{equation}
	s^2 = \frac{((x_{1} - \bar{x})^2 + (x_{2} - \bar{x})^2 + \cdots (x_{n} - \bar{x})^2)}{{n-1}} 
\end{equation}
\textbf{standard deviation} 
\begin{equation}
	s = \sqrt{s^2} 
\end{equation}
